	title = per.1.t
	desc = per.1.d

					}

							ideology = socialism
							popularity = 0.45
					}
			}
		
			IRQ = {
					add_popularity = {
							ideology = paternal_autocracy
							popularity = -0.40
					}

							ideology = neutrality
							popularity = 0.25
					}			
			}

			SYR = {
					add_popularity = {
							ideology = paternal_autocracy
							popularity = -0.25
					}
					add_popularity = {
							ideology = socialism
							popularity = 0.25
					}
			}		
	
		}
	}
	
	#big kurdistan#
	option = {
		name = per.1.a
		KUR = {
		transfer_state = 676

		}

		ai_chance = {
		factor = 55
		}
	}
	
	#give them the iraqi kurdistan#
	option = {
		name = per.1.b
		KUR = {
		transfer_state = 676
		}
		country_event = {id = per.2 days = 256}

		}
	}
