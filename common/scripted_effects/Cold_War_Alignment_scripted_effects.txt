replace_alignment_to_SOV = {
	
	if = {
		limit = {
		has_idea = usa_aligned_nation
		}
		swap_ideas = {
			remove_idea = usa_aligned_nation
			add_idea = ussr_aligned_nation
		}
	}
	else = {
		limit = {
		has_idea = china_aligned_nation
		}
		swap_ideas = {
			remove_idea = china_aligned_nation
			add_idea = ussr_aligned_nation
		}
	}
	else_if = {
		limit = {
			NOT = {
				OR = {
				has_idea = china_aligned_nation
				has_idea = usa_aligned_nation
				}
			}
		}
		add_ideas = ussr_aligned_nation
	}

}
